\section{The calculus}


\begin{frame}{The calculus $\Calc$}

  \small

  \begin{itemize}
  \item 
    $\Calc$ is inspired by a calculus %$\TKGL$
    for $\KGL$ presented in
  \citebib{B{\'{\i}}lkov{\'{a}} et al., IJCAR, 2022}.
  \begin{quote}
  \medfont  $\KGL$: extension of     $\GWCL$, over a more expressive language\\
(involutive negation,   co-implication)
\end{quote}
  


    

 \item 
   $\Calc$ is a generalization of the calculus %$\Calcc$
   for the logic $\GWCL$ introduced
   in  \citebib{Ferrari et al., TABLEAUX, 2025}.
  
  \end{itemize}
  

  
  \BSK

  \tb{$\Calc$ is a  refutation calculus} acting on constraints over
  labelled formulas (labels representing worlds of G\"odel-Kripke models).
  
  \MSK
  
  \bfb{Results overview}
  \begin{itemize}\small
  \item strong termination
  \item completeness
  \item countermodel-construction and finite model property
  \item proof-search procedure  %(no-backtracking)\\
  \item PSPACE-decidability 
  \item JTabWb implementation
  \end{itemize}
    
  \VFILL
\end{frame}

\begin{frame}
    \frametitle{The calculus $\Calc$: constraints}


%  \small
    Let  $\G$ be a set of constraints and  $\M=\stru{W,R,e}$  a model.

 \begin{center}
    \begin{tcolorbox}[colframe=blue!75!black,width=30em]
      \center
    \bfb{$\Mcal\models \G$  ($\Mcal$ satisfies $\G$)} iff:
      \begin{itemize}
      \item  there exists an  interpretation
  $\iota$  mapping each world label in $\G$ to a world in $W$
  such that
  $\iota(u)\abstractorder\iota(t)$ for every constraint $u\abstractorder t\in\G$\\    
        
\end{itemize}
\emm{All the constraints are  satisfied in  $\M$ under $\iota$} 
    \end{tcolorbox}
  \end{center}
    

  \MSK

  \tb{\G}\;=\;\{\; \ycbox{$w:\Box p \;\tr{<}\; w':q$}\,,\;\ggbox{$w':p\land q \;\tr{>}\; w:p$}\;\}
  \]

\begin{frame}
  \frametitle{The calculus $\Calc$: soundness}
  

  \bfb{Soundness}  
  \begin{center}
    \begin{tcolorbox}[colframe=blue!75!black,width=30em]
      \center
      $
      \proves{\Calc}{}{\G}
      \quad \tr{\Rightarrow}\quad
      \mbox{there is no $\GWL$-model $\M$ s.t.}~\M\models\G
      $
      \\[1ex]
      $\scriptsize \proves{\Calc}{}{\G} \quad {\red \rightsquigarrow} \quad \mbox{there exists a $\Calc$-derivation of $\G$}$
    \end{tcolorbox}
  \end{center}

  
  \SP

  {\small
    \begin{tcgreenbox}{Application}
    \vspace{-2ex}
    \[
    \begin{array}{c}
      \proves{\Calc}{}{\ycbox{$w_0:\varphi<1$}}
      \\[2ex] 
      \tr{\Downarrow}
      \\[1ex] 
      \mbox{there is no $\M$ s.t. $\M\models \ycbox{$w_0:\varphi <1$}$}
\quad  \text{\blue\em\scriptsize \emph\ no $\GWL$-countermodel for $\varphi$}
      \\[2ex]
        \tr{\Downarrow}
        \\[1ex] 
      \tb{\forall\,\M\,,\,\forall\,w\in\M}~e(w,\varphi)=1
      \\[1ex]
        \tr{\Downarrow}
        \\[1ex]
        \varphi\in\GWL
    \end{array}
    \]
    \vspace{.5ex}
  \end{tcgreenbox}
  }

  
  
  \VFILL
\end{frame}

\begin{frame}
  \frametitle{The calculus $\Calc$: some rules}


  Rules of the calculus, read bottom-up, must decompose a non-atomic constraint in a sound way

  \medskip
  {\blue\bf Rules for $\land$-constraints}

   \begin{tcolorbox}[colframe=blue!75!black,width=33em]
      \center
$e(w,\, \a \,\tr{\land}\, \b)\;=\; e(w,\a)\;\tr{\land}\;e(w,\b)\;=\;\tr{\min}\{\; e(w,\a)\,,\,e(w,\b)    \;\}$
\end{tcolorbox}   
    
  
  \[\small
    \begin{array}{c}
      \small
      %% AND >
    \AXC{$\ycbox{$w: \a \;\tm{>}\; t$},\;  \ycbox{$w: \a \;\tm{>}\; t$},   \, \G$}   
    % ------------------------------------------      
    \RightLabel{$\tr{\land >}$}
    \UIC{$\ggbox{$w: \a\land \b\; \tm{>}\; t$},\,\G$}
    \DP
      \\[4ex]
    e(w,\a \land \b) > t \quad\tr{\Implies}\quad  e(w,\a) > t\;\tr{\land }\;  e(w,\b) > t
      \\[6ex]
  %=============================================================    
    %% AND < 
    \AXC{$\ycbox{$w: \a \;\tm{<}\; t$},\, \G$}   
    \AXC{$\ycbox{$w: \b \;\tm{<}\; t$},\, \G$}
    % ------------------------------------------      
    \RightLabel{$\tr{\land <}$}
    \BIC{$\ggbox{$w: \a\land \b\; \tm{<}\; t$},\,\G$}
      \DP
        \\[4ex]
    e(w,\a \land \b) < t \quad\tr{\Implies}\quad  e(w,\a) > t\;\tr{\lor }\;  e(w,\b) > t
      \end{array}
    \]

    \smallskip
    The rules concerning the other propositional connectives are similar 
\end{frame}

\begin{frame}
  \frametitle{The calculus $\Calc$: some rules}
  
 {\blue\bf Rule for $\Box$-constraints}

 \begin{tcolorbox}[colframe=blue!75!black,width=33em]
      \center
 $ \small
 e(w, {\Box} \a) = {\tr\inf}\ \Big\{\,R(w,w')\,{\to}\, e(w',\a)~|~w'\in W\,\Big\}
 \qquad \tr{\inf = \min}$
\end{tcolorbox}
 
 \begin{itemize}
 \item \small

 
 \smallskip
 Let   $e(w, \Box \a)=\tm r$. The value $\tm r$ is \tb{witnessed} by a world $\tb{w_1}$, namely: 
\[\small
  \begin{array}{c}
    R(w,\tb{w_1}) \to \tb{w_1}: \a \,=\,\tm r
    \\[3ex]
    %% BOX LT
    \AXC{$\ycbox{$\overbrace{R(w,\tb{w_1}) \to \tb{w_1}: \a}^{\tm r} \;\tm{\leq}\; t$}\;,\;  \ycbox{$\Phibd{\G,w,\tb{w_1}}$}\;,   \; \G$}   
    % ------------------------------------------      
    \RightLabel{$\Box\tm{\leq}$}
    \UIC{$\ggbox{$\underbrace{w: \Box \a}_{\tm r}\; \tm{\leq}\; t$},\,\G$}
    \DP\qquad \text{\tb{$w_1$ new}}
    \\[10ex]
    \scriptsize
       \begin{array}{ccl}
         \Phibd{\G,w, w_1} &\,=\,&
                                   \{\, R(w,w_1)\to w_1: \b \gt\!' t~|~w :\Box \b\gt\!' t \in\G  \,\}\,\cup\,
                                   \\[1ex]
   \text{\scriptsize  constraints on $w_1$ induced by}      &&\{\, R(w,w_1)\land w_1: \b \lt\!' t~|~w :\Diam \b\lt\!' t \in\G  \,\}
          \\[1ex]
    \text{\scriptsize  modal constraints on  $w$}       && \lt,\lt'\in\{<,\leq\},\; \gt,\gt'\in\{>,\geq\}.
       \end{array}
  \end{array}
    \]
Without the witnessed condition, the rule is not sound.

\item
       No rule for constraints of the form $w: \Box \a > t$ and  $w: \Box \a \geq t$
      
\end{itemize}

     \end{frame}

\begin{frame}
  \frametitle{The calculus $\Calc$: axiom rule}\small



  \[\small
     %% AXAT 
    \AXC{}
    \RightLabel{\small$\ruleAxat$}
    \UIC{$\G$}    
    \DP  %\quad\mbox{if $\Atmp{\G}$ is not consistent}
\hspace{4em} \text{\tb{Soundness condition:}}\qquad \M\not\models \G,\quad \forall\,\M
\quad\text{(no countermodel)}
\]

\MP
  \begin{tcgreenbox}[top=0ex]{}
    A set of atomic constraint \bfb{$\Gat$ is consistent} iff there exists
     a  \tr{substitution $\s$} mapping
    \[
      \tb{\fbox{$w:p$}}\;\;\tr{\mapsto}\;\; r\in\Qrange
      \qquad
      \tb{\fbox{$R(w,w')$}}\;\;\tr{\mapsto}\;\; r\in\Qrange
      \qquad
      \tb{w:\bot}\;\;\tr{\mapsto}\;\; 0
    \]
    such that \tr{$\s$ is a solution to $\G$  (all the constraints in $\s(\Gat)$ are  satisfied)}.
  \end{tcgreenbox}

  \SSK

  \bfb{Example}
\small%\medfont
\[\medfont
    \Gat= \{\;\ycbox{$w_1 : p$} > \gcbox{$R(w_1,w_2)$}\,,\; \ggbox{$w_2 : p$} \leq \gcbox{$R(w_1,w_2)$}\,,\;
    \gcbox{$R(w_1,w_2)$} < 1\,,\;  \ggbox{$w_2 : p$} > \underset{\tm 0}{w_1 :\bot}   \;\}
  \]
Solution $\tm{\s}$ to $\Gat$:
    \[\medfont
      \begin{array}{c}
      \ycbox{$w_1:p$}\,\mapsto\, 0.8\,,\quad
    \gcbox{$R(w_1,w_2)$}\,\mapsto 0.6\,\,,\quad
    \ggbox{$w_2:p$}\,\mapsto 0.4
\\[4ex]
\scriptsize
        \s(\Gat) 
\\[-2ex]
\rotatebox{-90}{$\tr{\mapsto}$} 
\\[3ex]
        \tiny
        \s(\underset{\tm{0.8}}{\ycbox{$w_1 : p$}}) > \s(\underset{\tm{0.6}}{\gcbox{$R(w_1,w_2)$}})\,,\;
      \s(\underset{\tm{0.4}}{\ggbox{$w_2 : p$}}) \leq \s(\underset{\tm{0.6}}{\gcbox{$R(w_1,w_2)$}})\,,\;
       \s( \underset{\tm{0.6}}{\gcbox{$R(w_1,w_2)$}} < 1)\,,\;
       \s( \underset{\tm{0.4}}{\ggbox{$w_2 : p$}}) > \s(\underset{\tm 0}{w_1 :\bot})   
\end{array}
        \]

    


  \VFILL
\end{frame}

\begin{frame}{The calculus $\Calc$: axiom rule}
  \small
  \[
     %% AXAT 
    \AXC{}
    \RightLabel{\small$\ruleAxat$}
    \UIC{$\G$}    
    \DP\qquad\mbox{\tb{if $\Atmp{\G}$ is not consistent}}
 \]
%The side condition guarantees that there is no model $\M$ such that $\M\models \G$.
  
 \[
   \begin{array}{l}
    \mbox{\medfont $\tb{\Atmp{\G}}\;=\;\Atm{\G}\;\bigcup\;
      \underbrace{ %
        \{\;\tb{1} > t\,|\,\tb{\fbox{$w : \Box \a$}} > t \in \G\;\}
        \;\cup\;
        \{\;\tm{0} < t\,|\,\tm{\fbox{$w : \Diam \a$}} < t \in \G\;\}
     }_{\mbox{needed to guarantee the coherence}}$}
     \\[2ex]
\scriptsize     \Atm{\G}\quad {\red \rightsquigarrow} \quad \text{atomic constraints in $\G$}
   \end{array}  
   \]




\SP
  \bfb{Example}
  \[
    \begin{array}{rcl}
      \G &=&\{\;\tb{\fbox{$w_0 : \Box p_0$}} > w_1:p_1\,,\; w_1:p_1 \leq 1\,,\; w_1:p_1 \geq 1\;\}
   \\[2ex]          
      \Atmp{\G}& =  & \{\;\tb{1} > \ggbox{$w_1:p_1$}\,,\;\ggbox{$w_1:  p_1$} \leq 1\,,\; \ggbox{$w_1:p_1$} \geq 1\;\}
\qquad\text{\red not consistent}
    \end{array}
  \]
 


  \MP

  \begin{itemize}
  \item 
  
 Consistency of $\Atmp{\G}$  can be checked by a \tb{Constraint Solver
  over $\mathbf{Q}$}.

\end{center}

\medskip
\[
    \Atmp{\G} \quad\tr{\leadsto}\quad   \{\;{1} > \ggbox{$X$}\,,\;\ggbox{$X$} \leq 1\,,\; \ggbox{$X$} \geq 1\;\}
\qquad\text{\tr{no solution}}
  \]
\end{itemize}
