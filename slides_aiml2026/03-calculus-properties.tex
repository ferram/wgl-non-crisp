\section{Properties of the calculus}

  
\begin{frame}{Properties of $\Calc$}
  \small




  
  \begin{itemize}
  \item[(1)] 
 $\Calc$ is    {\red (strongly) terminating}, namely:
  
\begin{center}
    \begin{tcolorbox}[colframe=blue!75!black]
\small
     
    There exists a {\red well-founded relation} $\prec$ such that, for every application
    \[
\AXC{$\dots\qquad \sigma'\qquad \dots$}
\UIC{$\s$}
\DP
\]
of a  rule of $\Calc$, it holds that $\s'\prec \s$.
\end{tcolorbox}
 
\end{center}
   
Accordingly, backward proof search in $\Calc$  always terminates,
  whatever strategy ones applies. 


\item[(2)] 
$\Calc$ is a   {\red  sound} refutation calculus


  \begin{center}
    \begin{tcolorbox}[colframe=blue!75!black,width=22em]
       \center
  $
\proves{\Calc}{}{\G}
\qquad{\red \Implies}\qquad
\M\not\models \G,\ \forall\, \M
$
\[\scriptsize
  \begin{array}{rcl}
    \proves{\Calc}{}{\G}    & \red \rightsquigarrow & \mbox{there exists a $\Calc$-derivation of $\G$}
    \end{array}
\]
 \end{tcolorbox}
\end{center}


\end{itemize}

  \VFILL
\end{frame}

%======================================================================

\begin{frame}{Properties of $\Calc$}



  \begin{itemize}


\item[(3)] 
$\Calc$ is a   {\red  complete} refutation calculus


  \begin{center}
    \begin{tcolorbox}[colframe=blue!75!black,width=22em]
       \center
  $
\nproves{\Calc}{}{\G}
\qquad{\red \Implies}\qquad
\exists\ \M.\ \M\models \G
$
 \end{tcolorbox}
\end{center}

\end{itemize}
We have designed a {\red proof search strategy} for $\Calc$ which yields  {\red countermodels}
whenever proof search fails.


  Main idea:

  \begin{itemize}
  \item Bottom-up proof search, delaying as much as possible the application of modal rules

    \item If proof search fails, the  constraint set $\G'$ obtained at the end fully describes the countermodel.
  \end{itemize}

  

  \VFILL
\end{frame}

\begin{frame}{Properties of $\Calc$}

{\blue Example}


\begin{itemize}
\item   $\underbrace{\Box(p\lor q) \to (\Box p\lor \Diam q)}_{\varphi}$ (instance of the Crisp axiom) is not valid in $\GWL$

\item Proof-search for $\G=\{\,   w_0:\varphi < 1    \,\}$ fails

\item At the end we obtain the constraint set
  \[
    \begin{array}{lcl}
      \G' &=&\{\,  w_1:p<R(w_0,w_1),\;\; w_1:p\leq w_0 : q_0, \;\;  w_1 : q \geq R(w_0,w_1),
\\[.5ex] &&
\;\; 1> w_0 : q_0,   \;\;  R(w_0,w_1)\leq w_0 : q_0,\;
            \underline{w_0:  \Box(p\lor q)} > w_0 : q_0,\;\;
\\[.5ex] &&
          \;\;  \underline{w_0: \Diam q}  \leq  w_0 : q_0,\;\;  w_0 : q_0   < 1 \,\}
    \end{array}
    \]

 \item We compute  $\Atmp{ \G' }$:
\[            
  w_0:  \Box(p\lor q) \;\;\tm{\mapsto}\;\; 1 \hspace{4em}
  w_0: \Diam q\;\;\tm{\mapsto}\;\; 0
\]
\end{itemize}
\VFILL
\end{frame}

\begin{frame}{Properties of $\Calc$}

{\bf About Complexity}

  \smallskip

    The constraints set $\Atm{\G'}$ obtained at the and contains the description of the {\red whole} countermodel.

 As in the classical modal logic $\KL$, some formulas have countermodel of exponential size.

 Accordingly, the proof search procedure does not work in PSPACE
    

  
  However:

  \begin{theorem}
  The validity problem for $\GWL$ is PSPACE complete. 
  \end{theorem}

  \begin{itemize}
  \item We can define a (highly inefficient)  procedure to check if $\varphi\in\GWL$ which runs in polynomial space w.r.t.~the size of $\varphi$. 

  \item PSPACE-hardness follows from the fact that $\KL$ can be embedded into $\GWL$ (via the *-translation) and $\KL$ is PSPACE complete- 
  \end{itemize}



\end{frame}

\begin{frame}
  \frametitle{Complexity and implementation}
  \small
  
  \bfb{Complexity}\\
  The countermodel $\M$ for $w_0:\varphi<1$ has
  \begin{itemize}\small
  \item $depth\leq\size{\varphi}$
  \item every world of $\M$ has at most $\size{\varphi}$ R-successors
  \end{itemize}
  this implies that the size of $\M$ is $O(\size{\varphi}^{\size{\varphi}})$.


  % \MSK

  % It is not possible to build $\M$ one branch at a time, since the
  % constraints generated during the expansion of a branch have global
  % validity and must be kept throughout the construction.  

  \SSK
  
  By adapting the procedure described in
  \citebib{B{\'{\i}}lkov{\'{a}} et al.,2022} we can prove that the
  decision problem for $\GWCL$ is in PSPACE.

  \BSK

  \bfb{Implementation}: \bfb{gwcref}\\
  %%\tm{\small [\url{https://github.com/ferram/jtabwb_provers/tree/master/gwc_ref}]}
  Implementation of our proof-search procedure in JTabWb~\citebib{Ferrari et al., 2017}. 
  \begin{itemize}\small
  \item standard backward depth-first proof search and countermodel extraction
  \item consistency of atomic constraints is checked using the
    \tm{Choco-solver Java} library
  \item \LaTeX generation of proof-search trees and countermodels
  \end{itemize}

  \VFILL

\end{frame}
