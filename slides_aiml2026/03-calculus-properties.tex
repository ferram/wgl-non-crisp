\section{Properties of the calculus}

  
\begin{frame}{Properties of $\Calc$}
  \small




  
  \begin{itemize}
  \item[(1)] 
 $\Calc$ is    {\red (strongly) terminating}, namely:
  
\begin{center}
    \begin{tcolorbox}[colframe=blue!75!black]
\small
     
    There exists a {\red well-founded relation} $\prec$ such that, for every application
    \[
\AXC{$\dots\qquad \sigma'\qquad \dots$}
\UIC{$\s$}
\DP
\]
of a  rule of $\Calc$, it holds that $\s'\prec \s$.
\end{tcolorbox}
 
\end{center}
   
Accordingly, backward proof search in $\Calc$  always terminates,
  whatever strategy ones applies. 

\SP
\item[(2)] 
$\Calc$ is a   {\red  sound} refutation calculus


  \begin{center}
    \begin{tcolorbox}[colframe=blue!75!black,width=30em]
       \center
  $
\proves{\Calc}{}{\G}
\qquad{\red \Implies}\qquad
\M\not\models \G,\ \forall\, \M
\quad\text{(no countermodel)}
$
\[\scriptsize
  \begin{array}{rcl}
    \proves{\Calc}{}{\G}    & \red \rightsquigarrow & \mbox{there exists a $\Calc$-derivation of $\G$}
    \end{array}
\]
 \end{tcolorbox}
\end{center}


\end{itemize}

  \VFILL
\end{frame}

%======================================================================

\begin{frame}{Properties of $\Calc$}



  \begin{itemize}


\item[(3)] 
$\Calc$ is a   {\red  complete} refutation calculus


  \begin{center}
    \begin{tcolorbox}[colframe=blue!75!black,width=30em]
       \center
  $
\nproves{\Calc}{}{\G}
\qquad{\red \Implies}\qquad
\exists\ \M.\ \M\models \G
\quad\text{(exists a countermodel)}
$
 \end{tcolorbox}
\end{center}

% \smallskip
% \item[(4)] 
%   $\Calc$ can be easily turned into a   refutation calculus for $\GWCL$:\\
%   just add the constraints  $R(w,w')\in \{0,1\}$, for every label $w$, $w'$.



\end{itemize}

\MP

We have designed a {\red proof search strategy} for $\Calc$ which yields  {\red countermodels}
whenever proof search fails.

\medskip

\begin{tcgreenbox}[top=0ex]{Proof-search strategy (sketch)}
  \begin{itemize}
  \item[$\surd$] Apply rules bottom-up, delaying as much as possible the application of modal rules.


    \smallskip
  \item[$\surd$]  If proof search fails, the  constraint set $\G_{fin}$ obtained at the end
    fully describes the countermodel $\M$.
  \end{itemize}
 \end{tcgreenbox}


 \bigskip
 \bfb{Remark}

 \smallskip
  $\Calc$ can be  turned into a   refutation calculus for $\GWCL$:\\
   just add the constraints  $R(w,w')\in \{0,1\}$, for every label $w$, $w'$.
 

  \VFILL
\end{frame}

\begin{itemize}
\item Proof-search for $\G=\{\,   w_0:\varphi < 1    \,\}$ fails

 \medskip
\item In  the end we obtain the constraint set
  \[\small
    \begin{array}{lcl}
      \G_{fin} &=&\{\;  w_1:p<R(w_0,w_1)\,,\;\; w_1:p\leq w_0 : q_0\,, \;\;  w_1 : q \geq R(w_0,w_1)\,,

1> w_0 : q_0\,,   \;\;  R(w_0,w_1)\leq w_0 : q_0\,,\;
            \underline{w_0:  \Box(p\lor q)} > w_0 : q_0\,,
\\[.5ex] &&\;\;
   \underline{w_0: \Diam q}  \leq  w_0 : q_0\,,\;\;  w_0 : q_0   < 1 \;\}
    \end{array}
    \]


    \medskip
 \item We compute  $\Atmp{ \G_{fin} }$:
\[            
  w_0:  \Box(p\lor q) \;\;\tm{\mapsto}\;\; 1 \hspace{4em}
  w_0: \Diam q\;\;\tm{\mapsto}\;\; 0
\]
\end{itemize}

 \smallskip

  \begin{itemize}
  \item[$\surd$]   $\text{depth}(\M)\leq\size{\varphi}$ (size of $\varphi$)

  \item[$\surd$]  every world of $\M$ has at most $\size{\varphi}$ R-successors

  \item[$\surd$]  The codomains of $R$ and $e$ are \tr{finite} subsets   of $\Qrange$\\
\tr{(discrete countermodel)}
\end{itemize}

\smallskip
Accordingly, the \tr{size} of $\M$ is $\red O(\size{\varphi}^{\size{\varphi}})$.



\end{frame}

\begin{frame}{On complexity}
\small
\begin{theorem}
If $\varphi\not\in\GWL$,  there exists a countermodel of $\varphi$ of size  $O(\size{\varphi}^{\size{\varphi}})$.
  \end{theorem}

  \medskip
  There are formulas $\red \varphi$ having countermodels of size approximately  $\red \size{\varphi}^{\size{\varphi}}$.
  Accordingly,  the proof search procedure does not work in PSPACE (the constraint set
   obtained at the end of proof search contains the description of the {\red whole} countermodel).
  
  

\MP  
  However, reasoning as in   \citebib{B{\'{\i}}lkov{\'{a}} et al., IJCAR, 2022}, we can prove that:

  \begin{theorem}
  The validity problem for $\GWL$ is PSPACE complete. 
  \end{theorem}

  
  \begin{itemize}
    \medfont
  \item (Highly inefficient)  PSPACE-procedure to check if $\varphi\in\GWL$.

  \item PSPACE-hardness follows from the fact that:
    \begin{itemize}\medfont
    \item[$\surd$]    $\KL$ can be embedded into $\GWL$ (via the *-translation)

\item[$\surd$]  $\KL$ is PSPACE complete
    \end{itemize}


  \end{itemize}



\end{frame}
