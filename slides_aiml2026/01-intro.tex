
\section{The playground}


\begin{frame}{Fuzzy modal logics}
  
  
  In approximate reasoning it is usual \emb{to deal simultaneously
    with both fuzziness of propositions and modalities}, e.g., to
  assign a degree of truth to propositions like
  \begin{center}
    \emm{``John is possibly tall''} \quad or\quad \emm{``John is necessarily tall''}
  \end{center}
  where \emm{``John is tall''} is a \emb{fuzzy proposition} or address
  features like \emb{certainty}, \emb{belief} or \emb{similarity},
  which have natural interpretations in terms of modalities.

  \MSK

  \bfb{A natural semantics for fuzzy modal operators}\\
  Combine the \emb{Kripke semantics} for modal operators and one of the
  possible \emb{algebraic semantics} for many-valued logics.

  \MSK

  A preeminent choice for algebraic semantics is \emb{G\"odel
    algebra}, interpreting $x\land y$ with the continuous t-norm
  $\min\{x,y\}$ and $\to$ as its residuum, since\dots
  
    \SSK  

    \dots\emb{this is the only
    fuzzy logic whose modal analogue admits the normality axiom} $\Box
  (\a\to\b) \to (\Box\a\to\Box\b)$ \citebib{Buo et al., 2011}.

  
This beacuase this is the only fuzzy logic which admits weakening and contraction. 

  
  \VFILL
\end{frame}

\begin{frame}
  \frametitle{G\"odel-Kripke semantics}
  
  \bfb{Language}: $\tm{\PV}=\{p_1,p_2,\dots\}$, \tm{$\land$},
  \tm{$\lor$}, \tm{$\to$} ($\tm{\neg} \varphi=\varphi\to\bot$),
  \tb{$\Box$}, \tb{$\Diam$}.

  % \bfb{Formulas}
  % \begin{itemize}
  % \item countable set of propositional variables \tm{$\PV$}; 
  % \item propositional connectives: \tm{$\land$}, \tm{$\lor$}, \tm{$\to$} ($\tm{\neg}
  %   \varphi=\varphi\to\bot$)
  % \item modal operators: \tm{$\Box$}, \tm{$\Diam$}
  % \end{itemize}

  
  \MSK
  \bfb{G\"odel-Kripke model ($\GKL$-model)}: $\M=\stru{W,R,e}$ where:
  \begin{center}
  \begin{tabular}{p{15ex}p{25ex}p{20ex}}
    $W\not=\emptyset$&$\tm{R:W\times W\to [0,1]}$& $e:W\times\PV\to [0,1]$\\
    \mbox{\medfont worlds}&\mbox{\medfont \tm{accessibility relation}}&\mbox{\medfont evaluation}
  \end{tabular}
  \end{center}
  
  
  \begin{tcolorbox}[colframe=blue!75!black,width=\linewidth]
    \hspace{-2ex}\begin{tabular}{p{72ex}}
      \medfont $e(w,\bot) = 0$\\[2ex]
      \medfont $e(w,\a \tr{\star} \b) = e(w,\a)\, \tr{\star}\, e(w,\b)$, for 
      $\tr{\star}\in \{\,\land,\,\lor,\,\to\,,\,\neg\,\}$\\[1ex]
      \medfont$
      \quad a\tr{\land} b = \min(a,b)
      \quad
      a\tr{\lor} b = \max(a,b)
      \quad
      a\tr{\to} b =
      \begin{cases}
        \tb{1} & \mbox{if $a\leq b$}\\
        \tr{b} &  \mbox{if $a > b$}     
      \end{cases}$
      % \quad
      % \tr{\neg}a=%a\to\tr{\bot}=
      % \begin{cases}
      %   \tb{1} & \mbox{if {$a=0$}}\\     
      %   \tr{0} & \mbox{if {$a>0$}}
      % \end{cases}
      % $
      \\[4ex]
      \medfont $e(w,\tm{\Box} \a) = \tm{\inf}_{\tr{x}\in W}\Big\{\,R(w,\tr{x})\,\tr{\to}\, e(\tr{x},\a)\,\Big\}$
      \quad $e(w,\tm{\Diam} \a)= \tm{\sup}_{\tr{x}\in W}\Big\{\,R(w,\tr{x})\,\tr{\land}\, e(\tr{x},\a)\,\Big\}$
    \end{tabular}
  \end{tcolorbox}
  
  

  \begin{bbluebox}{}%{$\varphi$ valid in $\Mcal$ }%[colframe=blue!75!black][Subformula property]
    \begin{center}
      \bfb{$\varphi$ is valid in} $\tb{\Mcal}=\stru{W,R,e}$ iff \tb{$\forall w\in W$, 
      $e(w,\varphi)=1$}\qquad($\Mcal\models \varphi$)
    \end{center}
  \end{bbluebox}


  \VFILL

  \[
    \medfont
  \begin{array}{lcl}
    e(w_0,\Box p) &= &
                           \inf_{j\geq 0}\{\, \mbox{$R(w_0,w_j)\to  e(w_j,p)$} \,\}

    \\[1ex]
                  &\;=\; & \inf\,\{\,
       \underbrace{\overset{\tm{0}}{R(w_0,w_0)}\to  e(w_0,p)}_{\tm 1},\;
        \underbrace{\overset{\tm{1}}{R(w_0,w_1)}\to  \overset{\tm{1/2}}{e(w_1,p)}}_{\tm{1/2}},\;
          \underbrace{\overset{\tm{1}}{R(w_0,w_2)}\to  \overset{\tm{1/3}}{e(w_2,p)}}_{\tm{1/3}},\;\dots   \,\}
      \\[1ex]
       &\;=\; & \inf\,\{\, 1, \, 1/2,\, 1/3,\, 1/4, \, \dots \,\} \;=\; {\tm  0}                    
\end{array}
       \]

       \begin{itemize}
       \item \tm{0} is the infimum of  $R(w_0,w_j)\to  e(w_j,p)$  but \tm{not} the minimum!

\item Indeed, there is no $w_j$ such that $R(w_0,w_j)\to e(w_j,p) = 0$.
         
       \end{itemize}

\begin{frame}\frametitle{G\"odel modal logics}

  %\MSK
  \[
    \begin{array}{lcl}
 e(w,\Box \a) &= &\inf_{x\in W}\Big\{\,R(w,x)\,\to\, e(x,\a)\,\Big\}
 \\[1ex]
  e(w,\Diam \a) &= &\sup_{x\in W}\Big\{\,R(w,x)\,\land\, e(x,\a)\,\Big\}
    \end{array}
 \]

  
  {\bfb{Witnessed  G\"odel Modal Logic}} - \tm{$R$  witnessed}


 
  \begin{itemize}
  \item   $\M=\stru{W,R,e}$ is \tm{witnessed}  if the values of $e(w,\Box\a)$ and $e(w,\Diam\a)$
are witnessed by some world $w'$ in $\M$, namely:



 \begin{tcolorbox}[colframe=blue!75!black,width=\linewidth]
\[
\begin{array}{lcl}
 e(w,\Box \a)=r  &\tm\IFF& \exists\ w'\in W.\ R(w,w') \to  e(w',\a) = r
\\[.3ex]
&&\text{\small\tm{infimum}= \tm{minimum}}
  \\[1.5ex]
 e(w,\Diam \a)=r  &\tm\IFF& \exists\ w'\in W.\ R(w,w') \land  e(w',\a) = r
\\[.3ex]
&&\text{\small\tm{supremum}= \tm{maximum}}
\end{array}
\]
\end{tcolorbox}
Note that every finite model is witnessed

 
\end{itemize}



  \VFILL
  
\end{frame}
