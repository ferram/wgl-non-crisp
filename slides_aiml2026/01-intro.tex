
\section{The playground}


\begin{frame}{Fuzzy modal logics}
  
  
  In approximate reasoning it is usual \emb{to deal simultaneously
    with both fuzziness of propositions and modalities}, e.g., to
  assign a degree of truth to propositions like
  \begin{center}
    \emm{``John is possibly tall''} \quad or\quad \emm{``John is necessarily tall''}
  \end{center}
  where \emm{``John is tall''} is a \emb{fuzzy proposition} or address
  features like \emb{certainty}, \emb{belief} or \emb{similarity},
  which have natural interpretations in terms of modalities.

  \MSK

  \bfb{A natural semantics for fuzzy modal operators}\\
  Combine the \emb{Kripke semantics} for modal operators and one of the
  possible \emb{algebraic semantics} for many-valued logics.

  \MSK

  A preeminent choice for algebraic semantics is \emb{G\"odel
    algebra}, interpreting $x\land y$ with the continuous t-norm
  $\min\{x,y\}$ and $\to$ as its residuum, since\dots
  
    \SSK  

    \dots\emb{this is the only
    fuzzy logic whose modal analogue admits the normality axiom} $\Box
  (\a\to\b) \to (\Box\a\to\Box\b)$ \citebib{Bou et al., 2011}.

  
This beacuase this is the only fuzzy logic which admits weakening and contraction. 

  
  \VFILL
\end{frame}

\begin{frame}
  \frametitle{G\"odel-Kripke semantics}
  
  \bfb{Language}: $\tm{\PV}=\{p_1,p_2,\dots\}$, \tm{$\land$},
  \tm{$\lor$}, \tm{$\to$} ($\tm{\neg} \varphi=\varphi\to\bot$),
  \tb{$\Box$}, \tb{$\Diam$}.

  
  \MSK
  \bfb{G\"odel-Kripke model ($\GKL$-model)}: $\M=\stru{W,R,e}$ where:
  \begin{center}
  \begin{tabular}{p{15ex}p{25ex}p{20ex}}
    $W\not=\emptyset$&$\tm{R:W\times W\to [0,1]}$& $e:W\times\PV\to [0,1]$
    \\
    \mbox{\medfont worlds}&\mbox{\medfont \tm{accessibility relation}}&\mbox{\medfont evaluation}
  \end{tabular}
  \end{center}
  
\begin{tcolorbox}[colframe=blue!75!black,width=\linewidth]
    \hspace{-2ex}\begin{tabular}{p{72ex}}
      %\medfont
      $e(w,\bot) = 0$
      \\[2ex]
      %\medfont
      $e(w,\,\a \,\tr{\star} \,\b) = e(w,\a)\; \tr{\star}\; e(w,\b)$, for 
      $\tr{\star}\in \{\,\land,\,\lor,\,\to\,\}$
      \\[1ex]
      % \medfont
      $
      \quad a\,\tr{\land}\, b = \min(a,b)
      \qquad
      a\,\tr{\lor}\, b = \max(a,b)
      \qquad
      a\,\tr{\to}\, b =
      \begin{cases}
        \tb{1} & \mbox{if $a\leq b$}\\
        \tr{b} &  \mbox{if $a > b$}     
      \end{cases}$
      \\[4ex]
      \text{Note that}
      \\
  $ e(w, \tm{\neg}\a) =
      \begin{cases}
        \tb 1 & \mbox{if $e(w,\a)=0$}\\
        \tb 0 &  \mbox{if $e(w,\a) > 0$}     
      \end{cases}$
%       %\medfont
%       $e(w,\tm{\Box} \a) = \tm{\inf}_{\tr{x}\in W}\Big\{\,R(w,\tr{x})\,\tr{\to}\, e(\tr{x},\a)\,\Big\}$
% \\[2ex]
%       $e(w,\tm{\Diam} \a)= \tm{\sup}_{\tr{x}\in W}\Big\{\,R(w,\tr{x})\,\tr{\land}\, e(\tr{x},\a)\,\Big\}$
    \end{tabular}
  \end{tcolorbox}


  
  
  \VFILL

\begin{frame}
  \frametitle{G\"odel-Kripke semantics}


 $\tb{\M}=\stru{W,R,e}$ 
  
  \begin{tcolorbox}[colframe=blue!75!black,width=\linewidth]
    \hspace{-2ex}\begin{tabular}{p{72ex}}
         %\medfont
      $e(w,\tm{\Box} \a) \;=\; \tm{\inf}_{\tr{x}\in W}\Big\{\;R(w,\tr{x})\;\tr{\to}\; e(\tr{x},\a)\;\Big\}$
\\[4ex]
      $e(w,\tm{\Diam} \a)\;=\; \tm{\sup}_{\tr{x}\in W}\Big\{\;R(w,\tr{x})\;\tr{\land}\; e(\tr{x},\a)\:\Big\}$
    \end{tabular}
  \end{tcolorbox}
  
  
  \begin{bbluebox}{}%
    \[
      \begin{array}{lcl}
   \tb{\M\models \varphi}  &\;\tm\IFF\;& \forall w\in W,\ e(w,\varphi)=1
\\
\text{\medfont $\varphi$ is \tb{valid} in $\tb{\M}$}
      \end{array}
      \]
  \end{bbluebox}

  Assume that $\M\not\models \varphi$:

\smallskip
  \begin{itemize}
  \item there is $w\in w$ such that $e(w,\varphi) < 1$
  \end{itemize}
\smallskip
  
  We say that $\M$ is a \tb{countermodel} for $\varphi$.

  \VFILL

\end{frame}

\begin{frame}
  \frametitle{G\"odel modal logics: $\GKL$}

   \BSK
  {\bfb{G\"odel Modal Logic}} - \tm{no restriction on $R$}
 
  \[
  \GKL=\{~\varphi~|~\varphi~\mbox{valid in all G\"odel-Kripke models}~\}
  \]

  %\BSK
  \BSK
  \begin{itemize}
  \item  %[\tm{[1]}] 
$  \GKL\;\;\tm{=}\;\; \IPL \;\;\tm{+}\;\; (\a\to \b)\lor (\b\to\a) \;\;\tm{+} $  
\[\small
\begin{tabular}{m{1.5em}m{12em}m{1.5em}m{16em}}
  $(K_\Box)$ & $\Box(\a \to \b)\to (\Box\a\to \Box \b)$ & 
  $(K_\Diam)$ & $\Diam(\a \lor \b)\to (\Diam\a\lor \Diam \b)$
  \qquad $(F_\Diam)\; \neg \Diam\bot$ 
  \\[1ex]
  $(FS_1)$ & $\Diam(\a \to \b)\to (\Box\a\to \Diam \b)$ & 
  $(FS_2)$ & $(\Diam \a \to\Box \b)\to \Box(\a\to  \b)$
  \\[1ex]
  $(N_\Box)$ & $\vdash \a$ implies $\vdash \Box\a$ &
  $(N_\Diam)$ & $\vdash \a\to \b$ implies $\vdash \Diam \a\to\Diam\b$
\end{tabular}

    \small
  \begin{array}{lcl}
    e(w_0,\Box p) &= &
                           \inf\{\, R(w_0,w_j)\to  e(w_j,p)~|~w_j\in W \,\}

    \\[1ex]
                  &\;=\; & \inf\,\{\,
       \underbrace{\overset{\tm{0}}{R(w_0,w_0)}\to  e(w_0,p)}_{\tm 1},\;
        \underbrace{\overset{\tm{1}}{R(w_0,w_1)}\to  \overset{\tm{1/2}}{e(w_1,p)}}_{\tm{1/2}},\;\dots\,\}

\end{array}
       \]

       \begin{itemize}
       \item \tm{0} is the \tb{infimum} of the set of $R(w_0,w_j)\to  e(w_j,p)$,  but \tm{not} the minimum!

\item Indeed, there is no $w_j$ such that $R(w_0,w_j)\to e(w_j,p) = 0$.
         
       \end{itemize}

\begin{frame}\frametitle{G\"odel modal logics: witnessed models}

  %\MSK
  \[
    \begin{array}{lcl}
 e(w,\Box \a) &= &\tm{\inf}\ \Big\{\,R(w,x)\,\to\, e(x,\a)~|~x\in W \,\Big\}
 \\[2ex]
  e(w,\Diam \a) &= &\tm{\sup}\ \Big\{\,R(w,x)\,\land\, e(x,\a)~|~x\in W  \,\Big\}
    \end{array}
 \]

  
%  {\bfb{Witnessed  G\"odel Modal Logic}} - \tm{$R$  witnessed}


 
  \begin{itemize}
  \item   $\M=\stru{W,R,e}$ is \tm{witnessed}  if the values of $e(w,\Box\a)$ and $e(w,\Diam\a)$
are witnessed by some world $\tm{w'}$ in $\M$, namely:



 \begin{tcolorbox}[colframe=blue!75!black,width=\linewidth]
\[
\begin{array}{lcl}
 e(w,\Box \a)=r  &\tm\IFF& \exists\ \tm{w'}\in W.\ R(w,\tm{w'}) \to  e(\tm{w'},\a) = r
\\[.3ex]
        &&\tb{\text{\small infimum}  \,=\,  \text{\small minimum} }
  \\[1.5ex]
 e(w,\Diam \a)=r  &\tm\IFF& \exists\ \tm{w'}\in W.\ R(w,\tm{w'}) \land  e(\tm{w}',\a) = r
\\[.3ex]
                 &&
\tb{\text{\small supremum}  \,=\,  \text{\small maximum} }
 \end{array}
\]
\end{tcolorbox}


\item Every finite model is witnessed

 
\end{itemize}



  \VFILL
  
\end{frame}

\begin{frame}\frametitle{G\"odel modal logics: overview}

\input{fig-F5-logicsDiagram}

Let $\mathbf{L}$ be any of the above logics.

\begin{itemize}
\item In $\mathbf{L}$, $\Box$ and $\Diamond$ are not interdefinable

\item The classical modal logic $\KL$ can be embedded in $\mathbf{L}$:
  \[
    \varphi\in  \KL \qquad\tm{\IFF}\quad  \varphi^*\in \mathbf{L}
\qquad
          \begin{minipage}{10em}
            \begin{custombox}[top=0mm,bottom=0mm,width=\linewidth]
 \medfont             
$\begin{array}{ll}
  \varphi^* :&\ \forall p \text{ in }\varphi,
  \\[.5ex]
           &p\; \tm{\mapsto}\; \neg\neg p
\end{array} $
  \end{custombox}   
      \end{minipage}
    \]


\item decision problem for $\mathbf{L}$ is PSPACE complete
  \end{itemize}

  
\end{frame}
