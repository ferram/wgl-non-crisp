
\section{The playground}


\begin{frame}{Fuzzy modal logics}
  
  
  In approximate reasoning it is usual \emb{to deal simultaneously
    with both fuzziness of propositions and modalities}, e.g., to
  assign a degree of truth to propositions like
  \begin{center}
    \emm{``John is possibly tall''} \quad or\quad \emm{``John is necessarily tall''}
  \end{center}
  where \emm{``John is tall''} is a \emb{fuzzy proposition} or address
  features like \emb{certainty}, \emb{belief} or \emb{similarity},
  which have natural interpretations in terms of modalities.

  \MSK

  \bfb{A natural semantics for fuzzy modal operators}\\
  Combine the \emb{Kripke semantics} for modal operators and one of the
  possible \emb{algebraic semantics} for many-valued logics.

  \MSK

  A preeminent choice for algebraic semantics is \emb{G\"odel
    algebra}, interpreting $x\land y$ with the continuous t-norm
  $\min\{x,y\}$ and $\to$ as its residuum, since\dots
  
  \SSK  
  
  \dots\emb{this is the only
    fuzzy logic whose modal analogue admits the normality axiom} $\Box
  (\a\to\b) \to (\Box\a\to\Box\b)$ \citebib{Bou et al., 2011}.

  
  
  \VFILL
\end{frame}

\begin{frame}
  \frametitle{G\"odel-Kripke semantics}
  
  \bfb{Language}

  \medskip
  $\tm{\PV}=\{p_1,p_2,\dots\}$, \tm{$\land$},
  \tm{$\lor$}, \tm{$\to$},
  \tb{$\Box$}, \tb{$\Diam$}.

  
  \MSK
  \bfb{G\"odel-Kripke model ($\GKL$-model)}

\medskip
  $\M=\stru{W,R,e}$ where:
  \begin{center}
  \begin{tabular}{p{15ex}p{25ex}p{20ex}}
    $W\not=\emptyset$&$\tm{R:W\times W\to [0,1]}$& $e:W\times\PV\to [0,1]$
    \\
    \mbox{\medfont worlds}&\mbox{\medfont \tm{accessibility relation}}&\mbox{\medfont evaluation}
  \end{tabular}
  \end{center}

  \SP
\begin{tcolorbox}[colframe=blue!75!black,width=\linewidth]
    \hspace{-2ex}\begin{tabular}{p{72ex}}
      %\medfont
      $e(w,\tr\bot) = 0$
      \\[1ex]
      %\medfont
      $e(w,\,\a \,\tr{\star} \,\b) = e(w,\a)\; \tr{\star}\; e(w,\b)$, for 
      $\tr{\star}\in \{\,\land,\,\lor,\,\to\,\}$
      \\[1ex]
      % \medfont
      $
      \quad a\,\tr{\land}\, b = \min(a,b)
      \qquad
      a\,\tr{\lor}\, b = \max(a,b)
      \qquad
      \underset{\text{\tr{residuum}}}{a\,\tr{\to}\, b} =
      \begin{cases}
        \tb{1} & \mbox{if $a\leq b$}\\
        \tb{b} &  \mbox{if $a > b$}     
      \end{cases}$
         \end{tabular}
  \end{tcolorbox}

%\smallskip
%\medfont

\[\medfont%\scriptsize
\tm{\neg} \varphi  \coloneq \varphi\to\bot
\hspace{3em}
  e(w, \tm{\neg}\a) =
      \begin{cases}
        \tb 1 & \mbox{if $e(w,\a)=0$}\\
        \tb 0 &  \mbox{if $e(w,\a) > 0$}     
      \end{cases}
\]

  
  
  \VFILL
\end{frame}

\begin{frame}
  \frametitle{G\"odel-Kripke semantics}


% $\tb{\M}=\stru{W,R,e}$ 
  
  \begin{tcolorbox}[colframe=blue!75!black,width=\linewidth]
    \hspace{-2ex}\begin{tabular}{p{72ex}}
         %\medfont
      $e(w,\tm{\Box} \a) \;=\; \tm{\inf}_{\tr{w'}\in W}\Big\{\;R(w,\tr{w'})\;\tr{\to}\; e(\tr{w'},\a)\;\Big\}$
\\[4ex]
      $e(w,\tm{\Diam} \a)\;=\; \tm{\sup}_{\tr{w'}\in W}\Big\{\;R(w,\tr{w'})\;\tr{\land}\; e(\tr{w'},\a)\:\Big\}$
    \end{tabular}
  \end{tcolorbox}
  

  \SP
  \begin{bbluebox}{}%
    \[
      \begin{array}{lcl}
   \tb{\M\models \varphi}  &\;\tm\IFF\;& \forall w\in \M,\ e(w,\varphi)=1
\\
\text{\medfont $\varphi$ is \tb{valid} in $\M$}
      \end{array}
      \]
  \end{bbluebox}

  
  Assume that $\blue \M\not\models \varphi$:

  \smallskip
  \begin{itemize}
  \item there is a world $w$  in $\M$ such that $e(w,\varphi) < 1$
  \end{itemize}


  \MSK
  
  We say that $\M$ is a \tb{countermodel} for $\varphi$.

  \VFILL

\end{frame}

\begin{frame}
  \frametitle{G\"odel modal logics: $\GKL$}

   \BSK
  {\bfb{G\"odel Modal Logic}} - \tm{no restriction on $R$}

  
  \[
  \GKL=\{~\varphi~|~\varphi~\mbox{valid in all G\"odel-Kripke models}~\}
  \]

  \SP
    \begin{itemize}
  \item  %[\tm{[1]}] 
    $  \GKL\;\;\tm{=}\;\; \IPL \;\;\tm{+}\;\; $
    \begin{custombox}[top=0mm,bottom=0mm,width=33em]
\[\small
  \begin{tabular}{m{1.5em}m{12em}m{1.5em}m{16em}}
    \\[-3.5ex]
  \tb{(Lin)}  &   $(\a\to \b)\lor (\b\to\a)$         & $\tb{(F_\Diam)}\; \neg \Diam\bot$ 
\\[1ex]
  $\tb{(K_\Box)}$ & $\Box(\a \to \b)\to (\Box\a\to \Box \b)$ & 
  $\tb{(K_\Diam)}$ & $\Diam(\a \lor \b)\to (\Diam\a\lor \Diam \b)$
  \\[1ex]
  $\tb{(FS_1)}$ & $\Diam(\a \to \b)\to (\Box\a\to \Diam \b)$ & 
  $\tb{(FS_2)}$ & $(\Diam \a \to\Box \b)\to \Box(\a\to  \b)$
  \\[1ex]
  $\tb{(N_\Box)}$ & $\vdash \a$ implies $\vdash \Box\a$ &
  $\tb{(N_\Diam)}$ & $\vdash \a\to \b$ implies $\vdash \Diam \a\to\Diam\b$
\end{tabular}
\]
\end{custombox}

\citebib{\scriptsize Caicedo \& Rodr{\'{\i}}guez. Bi-modal {G}{\"{o}}del logic over
    [0,1]-valued Kripke frames. JLC  %\textbf{25}(1), 37--55 
    (2015)
}
    



\medskip
    \item
Decidable, PSPACE complete.

\smallskip
\citebib{\scriptsize
Caicedo et al. Decidability of order-based modal logics. J.~of Comp.~and System Sciences (2017).}


\medskip
    \item  %[\tm{[3]}]
Analytic calculi for the fragments   $\GKL_\Box$ and $\GKL_\Diamond$

\smallskip
\citebib{\scriptsize  Metcalfe \& Olivetti. Towards a proof theory of {G}{\"{o}}del modal
    logics. LMCS  % \textbf{7}(2)
    (2011)}
    
    \end{itemize}


 \VFILL

\end{frame}

 \begin{center}
    \input{fig-F2-nonWitnessed.tex}
  \end{center}
  
  \begin{tabular}{m{3em}m{0.5em}m{15em}}\small
    $e(w_0,\Box p)$ &= &\small
                           $\inf\{\, R(w_0,w_j)\to  e(w_j,p)~|~w_j\in W \,\}$
    \\[1ex]
                  &= & \medfont $\inf\,\{\,
       \underbrace{\overset{\tm{0}}{R(w_0,w_0)}\to  e(w_0,p)}_{\tm 1}\,,\;
        \underbrace{\overset{\tm{1}}{R(w_0,w_1)}\to  \overset{\tm{1/2}}{e(w_1,p)}}_{\tm{1/2}}\,,

\end{tabular}

\tr{not} the minimum!  (there is no $w_j$ such that $R(w_0,w_j)\to
e(w_j,p) = 0$)

\begin{frame}\frametitle{G\"odel modal logics: witnessed models}


  \begin{tcolorbox}[colframe=blue!75!black,width=30em]
  \[\small
    \begin{array}{lcl}
 e(w,\Box \a) &= &\tm{\inf}\ \Big\{\,R(w,w')\,\to\, e(w',\a)~|~w'\in W \,\Big\}
 \\[2ex]
  e(w,\Diam \a) &= &\tm{\sup}\ \Big\{\,R(w,w')\,\land\, e(w',\a)~|~w'\in W  \,\Big\}
    \end{array}
 \]
\end{tcolorbox}
  


\medskip
 

$\M=\stru{W,R,e}$ is \tm{witnessed}  if the values of $e(w,\Box\a)$ and $e(w,\Diam\a)$
are witnessed by some world $\tm{w'}$ in $\M$, namely:


\medskip
\begin{tcolorbox}[colframe=blue!75!black,width=30em]
  \[
  \begin{array}{lcl}
 e(w,\Box \a)=\tb{r}  &\tm\IFF& \exists\ \tm{w'}\in W.\ R(w,\tm{w'}) \to  e(\tm{w'},\a) = \tb{r}
\\[.3ex]
        &&\tb{\text{\small infimum}  \,=\,  \text{\small minimum} }
  \\[1.5ex]
 e(w,\Diam \a)=\tb{r}  &\tm\IFF& \exists\ \tm{w'}\in W.\ R(w,\tm{w'}) \land  e(\tm{w}',\a) = \tb{r}
\\[.3ex]
                 &&
\tb{\text{\small supremum}  \,=\,  \text{\small maximum} }
 \end{array}
\]
\end{tcolorbox}

\medskip
 Every finite model is witnessed




  \VFILL
  
\end{frame}

\begin{frame}\frametitle{G\"odel modal logics:  $\GWCL$ and  $\GWL$ }

  Witnessed models have been discusses in many papers

  \MSK

 
  \begin{itemize}
\item  In  \citebib{Ferrari et al.,  TABLEAUX 2025} we have introduced the logic
    \[
 \GWCL\;=\;\{~\varphi~|~\varphi~\mbox{valid in all}~
  \underbrace{\mbox{\bfm{witnessed crisp} G\"odel-Kripke models}}_{\mbox{\tb{$\GWCL$-models}}}~\}
      \]
      \begin{itemize}
      \item[$\surd$] refutation calculus for   $\GWCL$ (based on constraints)
 \item[$\surd$]  PSPACE completeness 
        \end{itemize}

      

\medskip
\item Here we extend the previous results to the logic      
  \[
  \GWL  =\{~\varphi~|~\varphi~\mbox{valid in all}~
  \underbrace{\mbox{\bfm{witnessed} G\"odel-Kripke models}}_{\mbox{\tb{$\GWL$-models}}}~\}
\]

  \begin{itemize}
      \item[$\surd$] refutation calculus for   $\GWL$ (based on constraints)
 \item[$\surd$]  PSPACE completeness 
        \end{itemize}

\end{itemize}

  \VFILL
  
\end{frame}

\begin{frame}\frametitle{G\"odel modal logics: overview}

\input{fig-F5-logicsDiagram}

Let $\mathbf{L}$ be any of the above logics.

\begin{itemize}
\item In $\mathbf{L}$, $\Box$ and $\Diamond$ are not interdefinable

\item The classical modal logic $\KL$ can be embedded in $\mathbf{L}$:
  \[
    \varphi\in  \KL \qquad\tm{\IFF}\quad  \varphi^*\in \mathbf{L}
\qquad
          \begin{minipage}{10em}
            \begin{custombox}[top=0mm,bottom=0mm,width=\linewidth]
 \medfont             
$\begin{array}{ll}
  \varphi^* :&\ \forall p \text{ in }\varphi,
  \\[.5ex]
           &p\; \tm{\mapsto}\; \neg\neg p
\end{array} $
  \end{custombox}   
      \end{minipage}
    \]


\item The decision problem for $\mathbf{L}$ is PSPACE complete
  \end{itemize}

  
\end{frame}
